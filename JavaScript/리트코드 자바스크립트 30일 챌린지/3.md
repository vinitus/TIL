```javascript
function createCounter(n) {
  let counter = n;
  return {
    increment() {
      counter += 1;
      return counter;
    },
    reset() {
      counter = n;
      return counter;
    },
    decrement() {
      counter -= 1;
      return counter;
    },
  };
}
```
